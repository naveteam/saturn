import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks'

import { Avatar } from '../components'
import { ThemeProvider } from '../theme'
import { iconNames } from '../utils'

<Meta title='Data Display/Avatar' argTypes={{
    variant: {
      control: {
        type: 'select',
        options: ['circle', 'square']
      }
    },
    fallback: {
      control: {
        type: 'select',
        options: iconNames
      }
    }
  }}
  component={Avatar}
/>

export const AvatarControl = args => (
  <ThemeProvider>
    <Avatar {...args} />
  </ThemeProvider>
)


# Avatar

O componente de `Avatar` é utilizado para mostrar o avatar de um usuário.

<Story
  name='Base'
  args={{
    avatar: 'https://source.unsplash.com/random/400x400'
  }}
  parameters={{
    design: {
      type: 'figma',
      url: 'https://www.figma.com/file/O3bKxIcsj2rc1FNIRclJyT/Saturn-System?node-id=915%3A0'
    }
  }}
>
  {AvatarControl.bind()}
</Story>

# Iniciais

Caso não seja passado um atributo `avatar` é possível passar as iniciais do nome do avatar, através da propriedade `letter`.

<Canvas>
  <Story
    name='Inicial'
    parameters={{
      design: {
        type: 'figma',
        url: 'https://www.figma.com/file/O3bKxIcsj2rc1FNIRclJyT/Saturn-System?node-id=915%3A0'
      }
    }}
  >
    <ThemeProvider>
      <Avatar letter='A' />
      <Avatar letter='AB' size='large' />
    </ThemeProvider>
  </Story>
</Canvas>

# Default

Caso não seja haja as propriedades `avatar` e `letter` será mostrado um ícone padrão para usuário e para empresa.
