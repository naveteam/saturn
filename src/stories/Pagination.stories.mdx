import { useState } from 'react'
import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks'

import { Pagination, SaturnProvider } from '../components'

<Meta title='Data input/Pagination' component={Pagination} />

export const PaginationControl = args => {
  const [page, setPage] = useState(1)
  return (
    <SaturnProvider>
      <Pagination {...args} page={page} onPageChange={setPage} />
    </SaturnProvider>
  )
}

export const PaginationPropsInput = () => {
  const [page, setPage] = useState(20)
  return <Pagination page={page} pageSize={200} onPageChange={setPage} variant='input' />
}

# Pagination

O componente de `Pagination` é utilizado para que o usuário possa trocar de página em sua aplicação. Sua estrutura é composta por um botões e dois ícones do tipo arrow.

# Pagination with Input

Caso haja mais de 100 páginas a prop `variant` deve ser passada. Neste caso a estrutura do componente será um input.
